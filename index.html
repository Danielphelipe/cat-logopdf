        // --- Estado Global ---

        let catalogItems = []; 

        let logoDataUrl = null;


        // --- Logo Handler ---

        logoInput.addEventListener('change', (e) => {

            const file = e.target.files[0];

            if (file) {

                const reader = new FileReader();

            handleFiles(e.dataTransfer.files);

        });

        dropZone.addEventListener('click', () => fileInput.click());

        // --- Core Logic ---


        function handleFiles(files) {

            if (!files.length) return;

            

            const newFiles = Array.from(files).filter(file => file.type.startsWith('image/'));

            

            if (newFiles.length === 0) {

                showStatus('Por favor, envie apenas arquivos de imagem (JPG/PNG).', 'error');

                return;

            }


            // Processar arquivos

            newFiles.forEach(file => {

                const data = parseFilename(file.name);

                catalogItems.push({

                    id: Date.now() + Math.random(),

                    file: file,

                    name: data.name,

                    color: data.color,

                    code: data.code,

                    isValid: data.isValid

                });

            });


            updateUI();

            showStatus(`${newFiles.length} novos produtos adicionados!`, 'success');

            

            // Scroll suave até a lista se for a primeira adição

            if (catalogItems.length === newFiles.length) {

                previewGrid.scrollIntoView({ behavior: 'smooth', block: 'start' });

            }

            

            fileInput.value = ''; // Reset input

        }


        function parseFilename(filename) {

            // Limpeza básica

            let clean = filename.replace(/\.[^/.]+$/, "").replace(/['"]+/g, "");

            

            let name = "Nome do Produto";

            let color = "Cor...";

            let code = "000";

            let isValid = false;


            // Lógica de extração: Nome - Cor [Código]

            const parts = clean.split('-');

            

            if (parts.length >= 2) {

                name = parts[0].trim();

                

                // O restante (cor + código)

                const rest = parts.slice(1).join('-').trim();

                

                // Busca código entre colchetes

                const matchCode = rest.match(/\[(.*?)\]/);

                

                if (matchCode) {

                    code = matchCode[1].trim();

                    // A cor é tudo antes do primeiro colchete

                    color = rest.split('[')[0].trim();

                    isValid = true;

                } else {

                    color = rest; // Assume tudo como cor se não achar código

                    code = "N/A";

                }

            } else {

                name = clean;

                isValid = false;

            }


            return { name, color, code, isValid };

        }

                // Habilitar botões de gerar

                desktopControls.classList.remove('opacity-50', 'pointer-events-none');

                catalogItems.forEach((item, index) => {

                    const reader = new FileReader();

                    reader.onload = (e) => {

                        const div = document.createElement('div');

                        // Usamos animate-fade-in para entrada suave

                        div.className = 'preview-card bg-white rounded-xl overflow-hidden relative group animate-fade-in flex flex-col h-full';

                        const statusIcon = item.isValid 

                            ? `<i class="fa-solid fa-check text-green-500" title="Padrão reconhecido"></i>` 

                            : `<i class="fa-solid fa-triangle-exclamation text-orange-400" title="Verifique o nome"></i>`;


                        div.innerHTML = `

                            <!-- Imagem Preview -->

        function updateItem(index, field, value) {

            if (catalogItems[index]) catalogItems[index][field] = value;

        }


        function removeItem(index) {

            catalogItems.splice(index, 1);

            updateUI();

        }


        function clearFiles() {

            if(confirm('Tem certeza que deseja remover todas as imagens?')) {

                catalogItems = [];

                fileInput.value = '';

                updateUI();

                showStatus('Lista limpa.', 'neutral');

            }

        }


        function showStatus(msg, type) {

            const colors = {

                success: 'bg-green-100 text-green-800 border-green-200',

                error: 'bg-red-100 text-red-800 border-red-200',

                neutral: 'bg-blue-50 text-blue-800 border-blue-200'

            };

            const icon = {

                success: '<i class="fa-solid fa-circle-check"></i>',

                error: '<i class="fa-solid fa-circle-exclamation"></i>',

                neutral: '<i class="fa-solid fa-info-circle"></i>'

            };


            statusMessage.className = `fixed bottom-8 left-1/2 transform -translate-x-1/2 z-50 px-6 py-4 rounded-full shadow-2xl flex items-center gap-3 font-bold text-sm ${colors[type] || colors.neutral} animate-fade-in`;

            statusMessage.innerHTML = `${icon[type] || icon.neutral} <span>${msg}</span>`;

            statusMessage.classList.remove('hidden');


            setTimeout(() => {

                statusMessage.classList.add('hidden');

            }, 4000);

        }

        async function generatePDF() {

            if (catalogItems.length === 0) return;


            const { jsPDF } = window.jspdf;

            

            // Configurações: 1300x1600px

            const doc = new jsPDF({ 

                orientation: 'portrait', 

                unit: 'px', 

                format: [1300, 1600]

            });


            // Metadados

            doc.setProperties({

                title: 'Catálogo Goulart Hair',

                author: 'Goulart Hair Marketing',

                creator: 'Gerador de Catálogo 3.0'

            });


            const pageWidth = 1300;

            const pageHeight = 1600;

            const margin = 80; // Margem generosa

            const brandColor = "#62311b"; 


            progressContainer.classList.remove('hidden');

            progressBar.style.width = '5%';

            progressBar.textContent = 'Iniciando...';


            try {

                // Preparar Logo

                let logoImg = null;

                let logoDims = { w: 0, h: 0 };

                

                if (logoDataUrl) {

                    try {

                        logoImg = await loadImage(logoDataUrl);

                        // Altura fixa de 140px para a logo no cabeçalho

                        const targetH = 140; 

                        const ratio = logoImg.width / logoImg.height;

                        logoDims.h = targetH;

                        logoDims.w = targetH * ratio;

                    } catch(e) { 

                        console.warn("Logo não carregada", e); 

                    }

                }


                for (let i = 0; i < catalogItems.length; i++) {

                    const item = catalogItems[i];

                    

                    if (i > 0) doc.addPage();

                    

                    // Fundo Branco (Segurança para impressão)

                    doc.setFillColor(255, 255, 255);

                    doc.rect(0, 0, pageWidth, pageHeight, 'F');


                    // --- 1. CABEÇALHO ---

                    const headerY = margin;

                    

                    // Logo (Esquerda)

                    if (logoImg) {

                        doc.addImage(logoDataUrl, 'PNG', margin, headerY, logoDims.w, logoDims.h);

                    }


                    // Título (Direita) - Alinhado verticalmente com o centro da logo

                    doc.setFont("helvetica", "normal");

                    doc.setFontSize(36);

                    doc.setTextColor(brandColor);

                    // Ajuste fino vertical para alinhar com logo

                    const textY = headerY + (logoDims.h ? logoDims.h / 1.5 : 50); 

                    doc.text("Catálogo Goulart hair", pageWidth - margin, textY, { align: 'right' });



                    // --- 2. RODAPÉ ---

                    const footerY = pageHeight - margin; // Linha base do rodapé

                    

                    // Layout do rodapé:

                    // Esquerda: Nome do Produto (Grande)

                    // Direita: Detalhes Técnicos (Alinhados à direita)


                    // Nome do Produto

                    doc.setFont("helvetica", "bold");

                    doc.setFontSize(60); 

                    doc.setTextColor(brandColor);

                    // Usar baseline 'bottom' facilita alinhar com a margem inferior

                    doc.text((item.name || "").toUpperCase(), margin, footerY);


                    // Detalhes Técnicos (Direita)

                    const detailsFontSize = 38;

                    const lineHeight = 55;

                    doc.setFontSize(detailsFontSize);

                    

                    // Código (Linha inferior, alinhada com o nome do produto)

                    doc.setFont("helvetica", "bold");

                    const codText = `Cód: ${item.code || ""}`;

                    doc.text(codText, pageWidth - margin, footerY, { align: 'right' });


                    // Cor (Linha superior)

                    const corText = `Cor: ${item.color || ""}`;

                    doc.text(corText, pageWidth - margin, footerY - lineHeight, { align: 'right' });



                    // --- 3. IMAGEM PRINCIPAL ---

                    // Calcular área livre exata

                    const headerBottom = headerY + (logoDims.h || 100) + 60; // Margem abaixo do header

                    const footerTop = footerY - lineHeight - 80; // Margem acima do footer (mais espaço para respiro)

                    

                    const availableW = pageWidth - (margin * 2);

                    const availableH = footerTop - headerBottom;


                    // Placeholder area de debug (opcional, remover em prod)

                    // doc.setDrawColor(200, 200, 200); doc.rect(margin, headerBottom, availableW, availableH); 


                    try {

                        let imgData = await readFileAsDataURL(item.file);

                        let img = await loadImage(imgData);


                        // Lógica de Rotação Automática (Se for paisagem, vira retrato)

                        if (img.width > img.height) {

                            const canvas = document.createElement('canvas');

                            canvas.width = img.height;

                            canvas.height = img.width;

                            const ctx = canvas.getContext('2d');

                            ctx.translate(canvas.width / 2, canvas.height / 2);

                            ctx.rotate(90 * Math.PI / 180);

                            ctx.drawImage(img, -img.width / 2, -img.height / 2);

                            imgData = canvas.toDataURL('image/jpeg', 0.95); // Alta qualidade

                            img = await loadImage(imgData); // Recarregar dimensões

                        }


                        // Lógica de Ajuste (Contain)

                        const imgRatio = img.width / img.height;

                        const areaRatio = availableW / availableH;

                        

                        let finalW, finalH;


                        if (imgRatio > areaRatio) {

                            finalW = availableW;

                            finalH = availableW / imgRatio;

                        } else {

                            finalH = availableH;

                            finalW = availableH * imgRatio;

                        }


                        // Centralizar

                        const x = margin + (availableW - finalW) / 2;

                        const y = headerBottom + (availableH - finalH) / 2;


                        doc.addImage(imgData, 'JPEG', x, y, finalW, finalH);


                    } catch (err) {

                        console.error("Erro ao processar imagem", err);

                    }


                    // Atualizar Barra

                    const percent = Math.round(((i + 1) / catalogItems.length) * 100);

                    progressBar.style.width = `${percent}%`;

                    progressBar.textContent = `${percent}%`;

                }


                doc.save('Catalogo_GoulartHair.pdf');

                showStatus('Download Concluído!', 'success');


            } catch (error) {

                console.error(error);

                showStatus('Erro ao gerar PDF. Tente novamente.', 'error');

            } finally {

                setTimeout(() => {

                    progressContainer.classList.add('hidden');

                    progressBar.style.width = '0';

                }, 3000);

            }

        }


        // --- Helpers ---

        const readFileAsDataURL = (file) => new Promise((resolve) => {

            const r = new FileReader();

            r.onload = (e) => resolve(e.target.result);

            r.readAsDataURL(file);

        });


        const loadImage = (src) => new Promise((resolve, reject) => {

            const i = new Image();

            i.onload = () => resolve(i);

            i.onerror = reject;

            i.src = src;

        });


    </script>
